function quickQC(qcId) {

  const qc = data.qc.find(q => q.id === qcId);

  if (!qc) return;

  const passQty = prompt(`工单 ${qc.woId} 抽检数量: ${qc.sampleQty}\n请输入合格数量:`);

  if (passQty === null) return;

  const pq = parseInt(passQty);

  if (isNaN(pq) || pq < 0 || pq > qc.sampleQty) {

    showToast('请输入有效的合格数量', 'error');

    return;

  }

  qc.passQty = pq;

  qc.result = pq >= qc.sampleQty * 0.95 ? 'pass' : 'fail';

  qc.appearance = qc.result === 'pass' ? 'pass' : 'fail';

  qc.adhesion = qc.result === 'pass' ? 'pass' : 'fail';

  if (qc.result === 'pass') {

    const wo = data.workOrders.find(w => w.id === qc.woId);

    if (wo) wo.status = 'completed';

  }

  e.preventDefault();

  const batch = document.getElementById('iqc-batch').value;

  const name = document.getElementById('iqc-name').value;

  const spec = document.getElementById('iqc-spec').value;

  const qty = parseInt(document.getElementById('iqc-qty').value);

  const sticky = parseInt(document.getElementById('iqc-sticky').value);

  const thickness = document.getElementById('iqc-thickness').value ? parseFloat(document.getElementById('iqc-thickness').value) : null;

  const remark = document.getElementById('iqc-remark').value;


  // SOP Logic Gate: Check QC data

  if (tension < 15) {

    showGateBlock(`进料批号 ${batch} 拉力值 ${tension}N 低于标准值 15N，系统拒绝入库！请安排复检或退货处理。`);

    return;

  }

  if (sticky < 3) {

    showGateBlock(`进料批号 ${batch} 黏性测试 ${sticky}级 低于合格标准 3级，系统拒绝入库！请联系供应商处理。`);

    return;

  }


  const now = new Date();

  const timeStr = `${now.getFullYear()}-${String(now.getMonth()+1).padStart(2,'0')}-${String(now.getDate()).padStart(2,'0')} ${String(now.getHours()).padStart(2,'0')}:${String(now.getMinutes()).padStart(2,'0')}`;

    id: batch, name, spec, qty, supplier, tension, sticky, thickness,

    status: 'stored', time: timeStr, remark

  });


  closeModal('iqcModal');

  document.getElementById('iqcForm').reset();

  renderAll();

  showToast(`原材料 ${batch} 已成功入库`, 'success');

}

function submitQC(e) {

  e.preventDefault();

  const type = document.getElementById('qc-type').value;

  const woId = document.getElementById('qc-wo').value;

  const sampleQty = parseInt(document.getElementById('qc-sample-qty').value);

  const passQty = parseInt(document.getElementById('qc-pass-qty').value);

  const appearance = document.getElementById('qc-appearance').value;

  const dimension = document.getElementById('qc-dimension').value ? parseFloat(document.getElementById('qc-dimension').value) : 0;

  const adhesion = document.getElementById('qc-adhesion').value;

  const remark = document.getElementById('qc-remark').value;


  // Determine result

  let result = 'pass';

  if (appearance === 'fail' || adhesion === 'fail' || Math.abs(dimension) > 0.5 || passQty < sampleQty * 0.95) {

    result = 'fail';

  }


  const now = new Date();

  const timeStr = `${now.getFullYear()}-${String(now.getMonth()+1).padStart(2,'0')}-${String(now.getDate()).padStart(2,'0')} ${String(now.getHours()).padStart(2,'0')}:${String(now.getMinutes()).padStart(2,'0')}`;

  const id = `QC-${String(data.qc.length + 1).padStart(3, '0')}`;


  data.qc.unshift({

    id, type, woId, sampleQty, passQty, appearance, dimension, adhesion, result, time: timeStr, remark

  });


  // If pass, update WO status to completed

  if (result === 'pass') {

    const wo = data.workOrders.find(w => w.id === woId);

    if (wo) wo.status = 'completed';

  }


  closeModal('qcModal');

  document.getElementById('qcForm').reset();

  renderAll();


  if (result === 'pass') {

    showToast(`检验合格！工单 ${woId} 已解锁，可安排出货`, 'success');

  } else {

    showToast(`检验不合格！工单 ${woId} 批次已锁定，禁止出货`, 'error');

  }

}


function submitShip(e) {

  e.preventDefault();

  const customer = document.getElementById('ship-customer').value;

  const productId = document.getElementById('ship-product').value;

  const qty = parseInt(document.getElementById('ship-qty').value);

  const logistics = document.getElementById('ship-logistics').value;

  const remark = document.getElementById('ship-remark').value;


  // SOP Gate: Check QC status

  const passedQC = data.qc.filter(q => q.result === 'pass').map(q => q.woId);

  if (!passedQC.includes(productId)) {

    showGateBlock(`成品批次 ${productId} 未通过品质检验，系统禁止出货！请先完成IPQC/OQC检验。`);

    return;

  }


  // Check inventory

  const wo = data.workOrders.find(w => w.id === productId);

  if (wo && qty > wo.actualQty) {

    showGateBlock(`出货数量 ${qty} 超过可用库存 ${wo.actualQty}，请调整出货数量！`);

    return;

  }


  const now = new Date();

  const dateStr = `${now.getFullYear()}${String(now.getMonth()+1).padStart(2,'0')}${String(now.getDate()).padStart(2,'0')}`;

  const id = `SH-${dateStr}-${String(data.shipping.length + 1).padStart(3, '0')}`;

  const timeStr = `${now.getFullYear()}-${String(now.getMonth()+1).padStart(2,'0')}-${String(now.getDate()).padStart(2,'0')} ${String(now.getHours()).padStart(2,'0')}:${String(now.getMinutes()).padStart(2,'0')}`;


  data.shipping.unshift({

    id, customer, product: productId, qty, logistics, time: timeStr, status: 'shipped', remark

  });


  // Deduct inventory

  if (wo) wo.actualQty -= qty;


  closeModal('shipModal');

  document.getElementById('shipForm').reset();

  renderAll();

  showToast(`出货单 ${id} 已确认，库存已扣减`, 'success');

}
